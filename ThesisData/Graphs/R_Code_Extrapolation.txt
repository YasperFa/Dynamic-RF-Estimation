# Read the datadata <- read.csv("/Users/sawsanfarah/Desktop/ThesisData/maxiumumVelocityInEachIteration/rayleighTaylor/dataMV.csv")# Extract rows 1 to 10 (skip the first point)first_10_excl_first <- data[1:10, ]# Perform linear regressionmodel <- lm(Maximum_Velocity ~ Iteration, data = first_10_excl_first)# Compute slope and prediction at iteration = 2000slope <- coef(model)["Iteration"]pred_2000 <- predict(model, newdata = data.frame(Iteration = 2000))# Plot the dataplot(first_10_excl_first$Iteration, first_10_excl_first$Maximum_Velocity,      main = "Linear Regression (Iterations 0â€“9)",     xlab = "Iteration", ylab = "Maximum Velocity",     xlim = c(0, 4000), ylim = c(0,100))# Create prediction line up to 4000new_iterations <- data.frame(Iteration = 1:4000)predicted_values <- predict(model, newdata = new_iterations)lines(new_iterations$Iteration, predicted_values, col = "#E69F00")# Add predicted point at iteration = 2000points(2000, pred_2000, col = "#009E73", pch = 19, cex = 1.2)text(2000, pred_2000, labels = paste0("Pred(2000) = ", round(pred_2000, 3)),      pos = 3, col = "#009E73")# Add slope as annotation in top-left cornerlegend("topleft", legend = paste("Slope =", round(slope, 6)), bty = "n")
